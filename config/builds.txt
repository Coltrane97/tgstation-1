Формат:
#build [имя билда] 
	dmbfile [имя] (имя *.dmb файла без расширения *.dmb)
	friendlyname [имя] (имя, показываемое игрокам)
	folder [имя] (имя папки, в котором находится билд)	
endbuild

build tgstation
	dmbfile tgstation
	friendlyname TG Station
	folder infinity
endbuild

build tgstationtest
	dmbfile tgstation
	friendlyname TG Station Test
	folder infinity-tg
endbuild

build paradise
	dmbfile paradise
	friendlyname Paradise Station
	folder infinity-par
endbuild

build paradisetest
	dmbfile paradise
	friendlyname Paradise Station Test
	folder infinity-paradise
endbuild
